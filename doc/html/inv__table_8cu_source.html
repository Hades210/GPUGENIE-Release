200 GPUGenie::inv_table::build(u64 max_length, bool use_load_balance)
201 {
202  _ck.clear(), _inv.clear();
203  _inv_index.clear();
204  _inv_pos.clear();
205  if(!use_load_balance)
206  {
207  max_length = (u64)0 - (u64)1;
208  }
209  unsigned int last;
210  int key, dim, value;
211  for (unsigned int i = 0; i < _inv_lists.size(); i++)
212  {
213  dim = i << _shifter;
214  for (value = _inv_lists[i].min(); value <= _inv_lists[i].max(); value++)
215  {
216  key = dim + value - _inv_lists[i].min();
217  vector<int>& index = *_inv_lists[i].index(value);
218 
219  if (_ck.size() <= (unsigned int) key)
220  {
221  last = _ck.size();
222  _ck.resize(key + 1);
223  _inv_index.resize(key + 1);
224  for (; last < _ck.size(); last++)
225  {
226  _ck[last] = _inv.size();
227  _inv_index[last] = _inv_pos.size();
228  }
229  }
230  for (unsigned int j = 0; j < index.size(); j++)
231  {
232  if (j % max_length == 0)
233  {
234  _inv_pos.push_back(_inv.size());
235  }
236  _inv.push_back(index[j]);
237  _ck[key] = _inv.size();
238  }
239 
240  }
241 
242  }
243  _inv_index.push_back(_inv_pos.size());
244  _inv_pos.push_back(_inv.size());
245 
246  _build_status = builded;
247  Logger::log(Logger::DEBUG, "inv_index size %d:", _inv_index.size());
248  Logger::log(Logger::DEBUG, "inv_pos size %d:", _inv_pos.size());
249  Logger::log(Logger::DEBUG, "inv size %d:", _inv.size());
250 }
