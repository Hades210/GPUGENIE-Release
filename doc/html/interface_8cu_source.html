200  std::vector<query>& queries, GPUGenie_Config& config)
201 {
202  queries.clear();
203  map<int, query> query_map;
204  int qid, dim, val;
205  float sel, weight;
206  for (unsigned int iq = 0; iq < config.multirange_query_points->size(); ++iq)
207  {
208  attr_t& attr = (*config.multirange_query_points)[iq];
209 
210  qid = attr.qid;
211  dim = attr.dim;
212  val = attr.value;
213  weight = attr.weight;
214  sel = attr.sel;
215  if (query_map.find(qid) == query_map.end())
216  {
217  query q(table, qid);
218  q.topk(config.num_of_topk);
219  if (config.selectivity > 0.0f)
220  {
221  q.selectivity(config.selectivity);
222  }
223  if (config.use_load_balance)
224  {
225  q.use_load_balance = true;
226  }
227  query_map[qid] = q;
228 
229  }
230  query_map[qid].attr(dim, val, weight, sel);
231  }
232  for (std::map<int, query>::iterator it = query_map.begin();
233  it != query_map.end() && queries.size() < (unsigned int) config.num_of_queries;
234  ++it)

236  query& q = it->second;
237  q.apply_adaptive_query_range();
238  queries.push_back(q);
