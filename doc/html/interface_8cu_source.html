924 void GPUGenie::sequence_to_gram(vector<vector<int> > & sequences, vector<vector<int> >& gram_data,
925  int max_value, int gram_length)
926 {
927  int num_of_value = max_value + 1;
928  int gram_value = 0;
929 
930  for(unsigned int i = 0 ; i < sequences.size() ; ++i)
931  {
932  int max_index = sequences[i].size() - gram_length + 1;
933  if(max_index <= 0)
934  {
935  vector<int> line_null;
936  vector<int> temp_line;
937  for(unsigned int p = 0; p < (unsigned int)gram_length; ++p)
938  {
939  if(p<sequences[i].size() <= 0) break;
940  if(p < sequences[i].size())
941  line_null.push_back(sequences[i][p]);
942  else
943  line_null.push_back(sequences[i][0]);
944 
945  }
946  int number = 0;
947  for(unsigned int p = 0; p < (unsigned int)gram_length; ++p)
948  {
949  number = number*num_of_value + line_null[p];
950  }
951  temp_line.push_back(number);
952  gram_data.push_back(temp_line);
953  continue;
954  }
955  vector<int> line;
956  for(unsigned int j = 0 ; j < (unsigned int)max_index ; ++j)
957  {
958  gram_value = 0;
959  for(unsigned int k = 0 ; k < (unsigned int)gram_length ; ++k )
960  {
961  gram_value = gram_value*num_of_value + sequences[i][j + k];
962  }
963  line.push_back(gram_value);
964  }
965  gram_data.push_back(line);
966  }
967 }
